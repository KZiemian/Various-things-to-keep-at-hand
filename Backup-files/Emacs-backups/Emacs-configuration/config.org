#+BEGIN_SRC emacs-lisp
  (autoload 'reftex-mode "reftex" "RefTeX Minor Mode" t)

  (add-hook 'LaTeX-mode-hook 'turn-on-reftex)
  (add-hook 'latex-mode-hook 'turn-on-reftex)

        TeX-newline-function 'reindent-then-newline-and-indent
